            {
                // PROJECT SETTINGS
                "editor.codeActionsOnSave": { // ESLint
                    // This will automatically correct all fixable ESLint errors, including the ones from the Prettier plugin:
                    "source.fixAll.eslint": "explicit"
                },
                "eslint.validate": [
                    "javascript",
                    "javascriptreact",
                    "vue"
                ],
                "editor.formatOnSave": true,

                // JAVASCRIPT SETTINGS
                "[javascript]": {
                    "editor.defaultFormatter": "dbaeumer.vscode-eslint", // ESLint
                },

                // VUE SETTINGS
                "[vue]": {
                    "editor.defaultFormatter": "dbaeumer.vscode-eslint", // ESLint
                },

                // CSS SETTINGS
                "[css]": {
                    "editor.defaultFormatter": "esbenp.prettier-vscode",
