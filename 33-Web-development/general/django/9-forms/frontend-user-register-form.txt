


FRONT-END: USER REGISTER FORM


    
    1) In a template (e.g. register.html) customize this:

        {% extends "base.html" %}
        {% block title %}Register{% endblock title %}
        {% block content %}
            <div class="row login text-center">
                <h1>Register</h1>
            </div>
            <div class="row">
                <div class="col"></div>
                <div class="login-form">
                    <!-- REGISTER FORM - START -->
                    <form method="post">
                        {% csrf_token %}
                        {{ register_form.as_p }}
                        <button type="submit">Register</button>
                        <a href="#">I already have an account</a>
                    </form>
                    <!-- REGISTER FORM - END --> 
                </div>
                <div class="col"></div>
            </div>     
        {% endblock content %}
