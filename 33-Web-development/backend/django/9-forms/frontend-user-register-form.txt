


FRONT-END: USER REGISTER FORM


    
    1) In a template (e.g. register.html) customize this:

        {% extends "base.html" %}
        {% block title %}Registro{% endblock %}
        {% block description %}{% endblock %}
        {% block content %}
            <div class="row login text-center">
                <h1>Register</h1>
            </div>
            <div class="row">
                <div class="col"></div>
                <div class="login-form">
                    <!-- REGISTER FORM - START -->
                    <form method="post">
                        {% csrf_token %}
                        {{ form.as_p }}
                        <button type="submit">Register now!</button>
                        <ul>
                            <li><a href="{% url 'accounts:login' %}">I already have an account</a></li>
                        </ul>
                    </form>
                    <!-- REGISTER FORM - END --> 
                </div>
                <div class="col"></div>
            </div>     
        {% endblock %}
